    import yfinance as yf
import pandas as pd
import os
from datetime import datetime


today_date = datetime.today().strftime('%Y-%m-%d')

# Get the stock data for the given ticker
stock = yf.Ticker(stock_name+ stockType)

# Get historical daily market data for the stock
hist = stock.history(period="max")

# Ensure data is not empty
if not hist.empty:
    # Calculate daily price change
    hist['Daily Change'] = hist['Close'].diff()  # Difference between consecutive close prices

    # Classify each day as an "up" or "down" day based on Daily Change
    hist['Trend'] = hist['Daily Change'].apply(lambda x: 'Up' if x > 0 else 'Down' if x < 0 else 'No Change')

    # Add columns to see high and low changes from the open price
    hist['High Change'] = hist['High'] - hist['Open']
    hist['Low Change'] = hist['Open'] - hist['Low']

    # Display the DataFrame with Ups and Downs data
    print(hist[['Open', 'High', 'Low', 'Close', 'Volume', 'Daily Change', 'Trend', 'High Change', 'Low Change']].head())

    
    # Save the DataFrame to a CSV file with the ticker symbol as part of the filename
    save_path = f"{stockFolder}{stock_name}_data_{today_date}.csv"

    hist.to_csv(save_path)
    print(f"Data saved to '{save_path}'")
else:
    print(f"No historical data found for {stock_name}.")

import pandas as pd
import numpy as np
import mplfinance as mpf
from sklearn.linear_model import LinearRegression
import os
from datetime import datetime
import matplotlib.pyplot as plt  # Importing matplotlib to display the figure

today_date = datetime.today().strftime('%Y-%m-%d')

plots_folder = os.path.join(stockFolder, 'plots')
os.makedirs(plots_folder, exist_ok=True)

file_path = os.path.join(stockFolder, latest_file)

# # Build the file path dynamically
# file_path = f"{stockFolder}{stock_name}_ups_downs.csv"

# Load the stock data
df = pd.read_csv(file_path)

# Ensure the 'Date' column is in datetime format
df['Date'] = pd.to_datetime(df['Date'])

# Set the 'Date' column as the index
df.set_index('Date', inplace=True)

# Ensure that the necessary columns are available: Open, High, Low, Close, Volume
df = df[['Open', 'High', 'Low', 'Close', 'Volume']]

# Prepare the data for Linear Regression (using 'Date' and 'Close' for prediction)
# Convert Date to numerical values for regression
df['Date_numeric'] = (df.index - df.index.min()) / np.timedelta64(1, 'D')

# Linear regression model to predict the closing price
X = df['Date_numeric'].values.reshape(-1, 1)  # Feature: Date_numeric
y = df['Close'].values  # Target: Close price
model = LinearRegression()
model.fit(X, y)

# Make predictions
df['Predicted_Close'] = model.predict(X)

# Plot the candlestick chart with predicted close prices
apds = [mpf.make_addplot(df['Predicted_Close'], color='red', linestyle='--')]

# Create the plot and save it directly as SVG
save_path = os.path.join(plots_folder, f"{stock_name}_stockData_{today_date}.svg")

# Save the plot
mpf.plot(df, type='candle', volume=True, style='charles', 
         title=f'{stock_name} Stock Price with Prediction', 
         ylabel='Price', ylabel_lower='Volume', addplot=apds,
         savefig=save_path)


print(f"Figure saved to {save_path}")

                   ### Using plotly (for interactive plots)

import pandas as pd
import plotly.graph_objects as go
import numpy as np
import os
from datetime import datetime
from scipy.stats import linregress

today_date = datetime.today().strftime('%Y-%m-%d')

plots_folder = os.path.join(stockFolder, 'plots')
os.makedirs(plots_folder, exist_ok=True)

latest_file = max([f for f in os.listdir(stockFolder) if f.startswith(f"{stock_name}_data_")], key=lambda f: datetime.strptime(f.split('_')[-1].split('.')[0], "%Y-%m-%d"))
file_path = os.path.join(stockFolder, latest_file)

# # Build the file path dynamically
# file_path = f"{stockFolder}{stock_name}_ups_downs.csv"

# Read the saved CSV file
df = pd.read_csv(file_path)

# Ensure the 'Date' column is in datetime format
df['Date'] = pd.to_datetime(df['Date'])

# Convert 'Date' to numeric values for regression
df['Date_numeric'] = (df['Date'] - df['Date'].min()) / np.timedelta64(1, 'D')

# Linear regression (Simple linear regression)
slope, intercept, r_value, p_value, std_err = linregress(df['Date_numeric'], df['Close'])
regression_line = slope * df['Date_numeric'] + intercept

# Polynomial regression (degree 3 for example)
coeffs = np.polyfit(df['Date_numeric'], df['Close'], 3)
poly = np.poly1d(coeffs)
smoothed_y_poly = poly(df['Date_numeric'])

# Create the figure using plotly.graph_objects
fig = go.Figure()

# Add candlesticks trace
fig.add_trace(go.Candlestick(
    x=df['Date'],
    open=df['Open'],
    high=df['High'],
    low=df['Low'],
    close=df['Close'],
    name=f"{stock_name} Candlesticks"
))

# Add the linear regression line to the plot
fig.add_trace(go.Scatter(
    x=df['Date'],
    y=regression_line,
    mode='lines',
    name='Linear Regression Line',
    line=dict(color='orange', width=2, dash='dash')
))

# Add the polynomial smoothed line to the plot
fig.add_trace(go.Scatter(
    x=df['Date'],
    y=smoothed_y_poly,
    mode='lines',
    name='Polynomial Smoothed Line',
    line=dict(color='green', width=2, dash='dot')
))

# Update layout to make it interactive
fig.update_layout(
    title=f"{stock_name} Stock Price Candlestick Chart with Best Fit Line",
    xaxis_title="Date",
    yaxis_title="Price (INR)",
    xaxis_rangeslider_visible=False,  # Remove range slider below the chart
    template="plotly_dark",  # Optional: Change to a dark theme
    xaxis_showgrid=True,
    yaxis_showgrid=True,
)


save_path = os.path.join(plots_folder, f"{stock_name}_stockDataWithBestFit_{today_date}.svg")

# Save the figure as a PNG file
fig.write_image(save_path)

print(f"Figure saved to {save_path}")

# Show the interactive plot
fig.show()

               data=response.json()

               for dataset in data['datasets']:
                   metric_name = dataset['metric']
                   values = dataset['values']
                   
                   # If values contain more than 2 items (like the 'delivery' field in 'Volume')
                   if len(values[0]) > 2:
                       # Separate the data into separate columns (Date, Value, Delivery)
                       processed_values = []
                       for entry in values:
                           date = entry[0]
                           value = entry[1]
                           delivery = entry[2]['delivery']  # Extract 'delivery' from the nested dictionary
                           processed_values.append([date, value, delivery])
                       
                       # Create a DataFrame for this metric with 'Date', 'Value', and 'Delivery' columns
                       metric_df = pd.DataFrame(processed_values, columns=['Date', f'{metric_name}_Value', f'{metric_name}_Delivery'])
                   else:
                       # Create a DataFrame for a simple metric (no nested data)
                       metric_df = pd.DataFrame(values, columns=['Date', f'{metric_name}_Value'])
                   
                   # Merge each metric's DataFrame on 'Date' to the combined DataFrame
                   if combined_df.empty:
                       combined_df = metric_df
                   else:
                       combined_df = pd.merge(combined_df, metric_df, on='Date', how='outer')
               
                       
               # Display the final combined DataFrame
               print(combined_df)
               
               save_path = f"{stockFolder}{stock_name}_data_{today_date}.csv"
               combined_df.to_csv(save_path, index=False)

               import matplotlib.pyplot as plt
               import pandas as pd
               import numpy as np
               import os
               from datetime import datetime
               
               # Assume the required variables are already set, e.g., combined_df, plots_folder, stock_name, today_date
               
               # Check if the 'Date' column is the index
               if 'Date' in combined_df.index.names:
                   # Reset the index to access 'Date' as a column if needed
                   combined_df.reset_index(inplace=True)
               
               # Convert 'Date' to datetime format, set it as the index again, and sort by date
               combined_df['Date'] = pd.to_datetime(combined_df['Date'])
               combined_df.set_index('Date', inplace=True)
               combined_df.sort_index(inplace=True)
               
               # Convert 'Price_Value' to numeric in case there are any non-numeric entries
               combined_df['Price_Value'] = pd.to_numeric(combined_df['Price_Value'], errors='coerce')
               
               # Drop any rows with NaN in 'Price_Value' if necessary
               combined_df.dropna(subset=['Price_Value'], inplace=True)
               
               # Plot 'Price_Value' over time with dates labeled at 3-month intervals
               plt.figure(figsize=(10, 6))
               plt.plot(combined_df.index, combined_df['Price_Value'], label='Price Value', color='black')
               
               # Set x-axis to display labels at 3-month intervals
               plt.gca().xaxis.set_major_locator(plt.matplotlib.dates.MonthLocator(interval=3))
               plt.gca().xaxis.set_major_formatter(plt.matplotlib.dates.DateFormatter('%Y-%m'))
               
               # Add X and Y axis lines
               plt.axhline(0, color='black',linewidth=0.5)  # Y-axis line at y=0
               plt.axvline(combined_df.index[0], color='black',linewidth=0.5)  # X-axis line at the first date
               
               # Linear Best Fit Line
               x_vals = np.arange(len(combined_df))  # Generate x values
               y_vals = combined_df['Price_Value'].values  # Get y values
               
               # Linear fit (1st degree polynomial)
               linear_fit = np.polyfit(x_vals, y_vals, 1)
               linear_line = np.poly1d(linear_fit)
               plt.plot(combined_df.index, linear_line(x_vals), label='Linear Fit', color='red', linestyle='--')
               
               # Polynomial (2nd degree) Best Fit Line
               polynomial_fit = np.polyfit(x_vals, y_vals, 2)
               polynomial_line = np.poly1d(polynomial_fit)
               plt.plot(combined_df.index, polynomial_line(x_vals), label='Polynomial Fit (2nd degree)', color='green', linestyle='-.')
               
               # Labeling
               plt.title(f"{stock_name} Stock Price Chart with Best Fit Line")
               plt.xlabel("Date")
               plt.ylabel("Price Value")
               plt.xticks(rotation=45)
               
               # Set y-axis to start from 20
               # plt.ylim(bottom=20)
               
               # Display grid and legend
               plt.grid()
               plt.legend()
               plt.tight_layout()
               
               # Save the plot
               save_path = os.path.join(plots_folder, f"{stock_name}_stockDataWithBestFit_{today_date}.svg")
               plt.savefig(save_path, format='svg')
               
               # Show plot
               plt.show()
