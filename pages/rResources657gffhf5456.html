               plot_list <- list()
               count<-0
               for (rowNum in 1:dim(reshaped_data)[1]){
                 
                 descrStr <- reshaped_data$selectedComparison[rowNum]
                 split_str <- strsplit(descrStr, " ")[[1]]
                 chan <- split_str[1]
                 erp <- split_str[2]
                 
               Group <- c("AlcAvoi", "AlcAppr", "NonAlc")
               DvValue <- c(reshaped_data$AlcAvoi_Mean[rowNum], reshaped_data$AlcAppr_Mean[rowNum], 
                           reshaped_data$NonAlc_Mean[rowNum])
               SE <- c(reshaped_data$AlcAvoi_SE[rowNum], reshaped_data$AlcAppr_SE[rowNum], 
                           reshaped_data$NonAlc_SE[rowNum])
               sigVal <- c(reshaped_data$AlcAvoi_vs_AlcAppr_p[rowNum], 
                           reshaped_data$NonAlc_vs_AlcAppr_p[rowNum], 
                           reshaped_data$NonAlc_vs_AlcAvoi_p[rowNum])
               sigVal <- round(sigVal, 3)
               sigLabel <- ifelse(sigVal <= 0.001, "***",
                                     ifelse(sigVal < 0.01, "**",
                                            ifelse(sigVal < 0.05, "*", "")))
               data <- data.frame(Group, DvValue, SE, sigVal, sigLabel)
               
                 Q <- ggplot(data, aes(x = Group, y = DvValue, fill = Group)) +
                   geom_bar(stat = "identity", width = 0.7, color = "black") +
                   geom_errorbar(aes(ymin = DvValue - SE, ymax = DvValue + SE), width = 0.2, color = "black") +
                   geom_signif(comparisons = list(c("AlcAvoi", "AlcAppr")), 
                               annotations = paste(as.character(data$sigVal[1]), data$sigLabel[1]), 
                               y_position = 4.5, tip_length = 0.03, textsize = 3) +
                   geom_signif(comparisons = list(c("NonAlc", "AlcAppr")), 
                               annotations = paste( as.character(data$sigVal[2]), data$sigLabel[2]), 
                               y_position = 5.8, tip_length = 0.03, textsize = 3) +
                   geom_signif(comparisons = list(c("NonAlc", "AlcAvoi")), 
                               annotations = paste(as.character(data$sigVal[3]), data$sigLabel[3]), 
                               y_position = 5.1, tip_length = 0.03, textsize = 3) +
                   geom_hline(yintercept = 0, linetype = "dashed", color = "black") +
                   ggtitle(paste(chan, erp)) +
                   labs(y = "Mean Amplitude (µV)", x = NULL) +
                   scale_fill_manual(values = c("gray10", "gray50", "gray95")) +
                   # scale_fill_manual(values = c("deepskyblue", "sienna1", "gold1")) +
                   scale_y_continuous(breaks = seq(-6, 6, by = 3), limits = c(-6, 6.5)) +
                   
                   # Clean and professional theme with Times New Roman font
                   theme_classic(base_size = 14) +
                   theme(
                     text = element_text(family = "Times New Roman"),  # Set font to Times New Roman
                     plot.title = element_text(hjust = 0.5, size = 12, face = "bold"),
                     axis.text.x = element_text(size = 12,angle = 45, hjust = 1),
                     axis.text.y = element_text(size = 12),
                     axis.title.y = element_text(size = 12),
                     legend.position = "none"  # Remove legend if redundant
                   ) +
                   coord_capped_cart(left = "both", bottom = "both") 
                 
                 count=count+1
                 plot_list[[count]] <- Q
                 
               }

    • Cohen's d
      •       compute_cohens_d_AlcStimGrpDif <- function(group1_data, group2_data, groups) {
         cohens_d_result <- cohen.d(group1_data, group2_data)
      
         # Compute stats for both groups
         group1_n <- sum(!is.na(group1_data))
         group1_mean <- mean(group1_data, na.rm = TRUE)
         group1_sd <- sd(group1_data, na.rm = TRUE)
         group1_se <- group1_sd / sqrt(group1_n)
      
         group2_n <- sum(!is.na(group2_data))
         group2_mean <- mean(group2_data, na.rm = TRUE)
         group2_sd <- sd(group2_data, na.rm = TRUE)
         group2_se <- group2_sd / sqrt(group2_n)
      
         return(data.frame(
            stim = "Alcohol",
            grp1 = groups[1],
            grp2 = groups[2],
            n1 = group1_n,
            Mean1 = group1_mean,
            SD1 = group1_sd,
            SE1 = group1_se,
            n2 = group2_n,
            Mean2 = group2_mean,
            SD2 = group2_sd,
            SE2 = group2_se,
            Cohen_D = cohens_d_result$estimate,
            Lower_CI = cohens_d_result$conf.int[1],
            Upper_CI = cohens_d_result$conf.int[2],
            Interpretation = cohens_d_result$magnitude
         ))
      }
